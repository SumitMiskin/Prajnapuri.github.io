Problem Statement

Analyze and rank city routes by congestion level so planners can prioritize decongestion measures. This implementation uses a binary heap (max-heap) to sort routes by congestion.

Solution Approach

The algorithm builds a max-heap from route congestion values and repeatedly removes the maximum to produce a sorted list. Route names are carried alongside congestion values so the final order lists routes from least to most congested.

This is useful for dashboard summaries, prioritized interventions, and quick filtering of critical routes.

5// Function to heapify from bottom up based on congestion


6void BuildMaxHeap(int congestion[], string routes[], int n) {


7    for (int i = n / 2; i >= 1; i--) {


8        int k = i;


9        int v = congestion[k];


10        string route = routes[k];


11        int heap = 0;


12


13        while (heap == 0 && 2 * k <= n) {


14            int j = 2 * k;


15            if (j < n && congestion[j] < congestion[j + 1]) {


16                j++;


17            }


18


19            if (v >= congestion[j]) {


20                heap = 1;


21            } else {


22                congestion[k] = congestion[j];


23                routes[k] = routes[j];


24                k = j;


25            }


26        }


27        congestion[k] = v;


28        routes[k] = route;


29    }


30}

32// Function to delete the highest congestion route (root of heap)


33void RemoveMax(int congestion[], string routes[], int &n) {


34    swap(congestion[1], congestion[n]);


35    swap(routes[1], routes[n]);


36    n--;


37    int k = 1;


38    int v = congestion[k];


39    string route = routes[k];


40    int heap = 0;


41


42    while (heap == 0 && 2 * k <= n) {


43        int j = 2 * k;


44        if (j < n && congestion[j] < congestion[j + 1]) {


45            j++;


46        }


47


48        if (v >= congestion[j]) {


49            heap = 1;


50        } else {


51            congestion[k] = congestion[j];


52            routes[k] = routes[j];


53            k = j;


54        }


55    }


56    congestion[k] = v;


57    routes[k] = route;


58}

60// Heap Sort based on traffic congestion


61void HeapSortRoutes(int congestion[], string routes[], int n) {


62    BuildMaxHeap(congestion, routes, n);


63    for (int i = n; i > 1; i--) {


64        RemoveMax(congestion, routes, n);


65    }


66}
