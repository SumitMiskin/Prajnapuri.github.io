Problem Statement

Students need to take courses in a specific valid order based on dependencies (prerequisites). Determining a correct study path becomes complex when there are many interlinked courses.

Solution Approach

We model the courses and prerequisites as a Directed Acyclic Graph (DAG) and use **Topological Sort** (via DFS). This algorithm linearly orders the courses such that for every dependency u -> v, course u comes before v in the ordering, ensuring a valid study plan.

C++ Implementation
1#include <iostream>


2#include <vector>


3#include <stack>


4using namespace std;


5


6class CourseScheduler {


7    int numCourses;


8    vector<vector<int>> adjList;


9    vector<bool> visited;


10    stack<int> topoStack;


11


12public:


13    CourseScheduler(int n) {


14        numCourses = n;


15        adjList.resize(n);


16        visited.resize(n, false);


17    }


18


19    // Add prerequisite relation (u -> v)


20    void addPrerequisite(int u, int v) {


21        adjList[u].push_back(v);


22    }


23


24    void dfs(int node) {


25        visited[node] = true;


26        for (int neighbor : adjList[node]) {


27            if (!visited[neighbor])


28                dfs(neighbor);


29        }


30        topoStack.push(node);


31    }


32


33    void topologicalSort() {


34        for (int i = 0; i < numCourses; i++) {


35            if (!visited[i])


36                dfs(i);


37        }


38


39        cout << "\nSuggested course order:\n";


40        while (!topoStack.empty()) {


41            cout << "Course " << topoStack.top() << endl;


42            topoStack.pop();


43        }


44    }


45};
