5// Check if two zones belong to the same group (to avoid cycles)


6int isConnected(int zoneGroup[], int a, int b) {


7    return zoneGroup[a] == zoneGroup[b];


8}


9


10// Merge two zone groups


11void connectZones(int zoneGroup[], int n, int a, int b) {


12    int oldGroup = zoneGroup[a];


13    for (int i = 0; i < n; i++) {


14        if (zoneGroup[i] == oldGroup)


15            zoneGroup[i] = zoneGroup[b];


16    }


17}

19// Partition function for QuickSort (sorts by road cost)


20int partition(int cost[], int from[], int to[], int l, int r) {


21    int pivot = cost[l];


22    int pivotFrom = from[l];


23    int pivotTo = to[l];


24    int i = l, j = r + 1;


25


26    while (true) {


27        while (i <= r && cost[++i] <= pivot);


28        while (cost[--j] > pivot);


29        if (i >= j) break;


30        swap(cost[i], cost[j]);


31        swap(from[i], from[j]);


32        swap(to[i], to[j]);


33    }


34


35    swap(cost[l], cost[j]);


36    swap(from[l], from[j]);


37    swap(to[l], to[j]);


38    return j;


39}

41void quickSort(int cost[], int from[], int to[], int l, int r) {


42    if (l < r) {


43        int p = partition(cost, from, to, l, r);


44        quickSort(cost, from, to, l, p - 1);


45        quickSort(cost, from, to, p + 1, r);


46    }


47}

49// Kruskal's MST algorithm adapted for smart roads


50void planSmartRoadNetwork(int n, int e, int from[], int to[], int cost[], string locationNames[]) {


51    int zoneGroup[n];


52    for (int i = 0; i < n; i++) zoneGroup[i] = i;


53


54    int totalCost = 0;


55    quickSort(cost, from, to, 0, e - 1);


56


57    cout << "\nOptimized Road Plan (Minimum Spanning Tree):\n";


58    for (int i = 0; i < e; i++) {


59        if (!isConnected(zoneGroup, from[i], to[i])) {


60            connectZones(zoneGroup, n, from[i], to[i]);


61            totalCost += cost[i];


62            cout << locationNames[from[i]] << " - " << locationNames[to[i]]


63                 << " (Cost: " << cost[i] << " mins)\n";


64        }


65    }


66    cout << "Total minimum infrastructure cost: " << totalCost << " mins\n";


67}
