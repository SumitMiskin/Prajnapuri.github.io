Problem Statement

Maintain and query vehicle counts across a series of traffic checkpoints efficiently, supporting point updates and prefix-sum queries for cumulative traffic measurements.

Solution Approach

We use a Fenwick Tree (Binary Indexed Tree) to keep cumulative counts with O(log n) updates and queries. This is suitable for dashboards and real-time aggregation of sensor data across checkpoints.

C++ Implementation


1#include <iostream>


2using namespace std;


3


4int n, vehicles[20], BIT[20];


5


6class TrafficMonitor {


7public:


8    TrafficMonitor(int n);


9    void initializeSensors();


10    void updateCheckpoint(int index, int delta);


11    int totalVehiclesUpTo(int checkpoint);


12    int getTotalTraffic();


13};


14


15TrafficMonitor::TrafficMonitor(int n) {


16    for (int i = 0; i <= n; i++) {


17        BIT[i] = 0;


18    }


19}


20


21void TrafficMonitor::initializeSensors() {


22    cout << "Enter vehicle count for each of the " << n << " checkpoints:\n";


23    for (int i = 1; i <= n; i++) {


24        cout << "Checkpoint " << i << ": ";


25        cin >> vehicles[i];


26        updateCheckpoint(i, vehicles[i]);


27    }


28}


29


30void TrafficMonitor::updateCheckpoint(int index, int delta) {


31    for (; index <= n; index += index & -index) {


32        BIT[index] += delta;


33    }


34}


35


36int TrafficMonitor::totalVehiclesUpTo(int checkpoint) {


37    int sum = 0;


38    while (checkpoint > 0) {


39        sum += BIT[checkpoint];


40        checkpoint -= checkpoint & -checkpoint;


41    }


42    return sum;


43}


44


45int TrafficMonitor::getTotalTraffic() {


46    return totalVehiclesUpTo(n);


47}
