56void SmartTrafficNetwork::computeAllShortestRoutes() {


57    for (int k = 1; k <= numLocations; k++) {


58        for (int i = 1; i <= numLocations; i++) {


59            for (int j = 1; j <= numLocations; j++) {


60                if (travelTime[i][k] != 999 && travelTime[k][j] != 999 &&


61                    (travelTime[i][k] + travelTime[k][j] < travelTime[i][j])) {


62                    travelTime[i][j] = travelTime[i][k] + travelTime[k][j];


63                }


64            }


65        }


66    }


67}
