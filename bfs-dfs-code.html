Problem Statement

Analyze city zone connectivity to find all reachable zones, detect isolated regions, and explore alternate paths during traffic congestion or emergencies.

Solution Approach

BFS (Breadth-First Search): Explores layer-by-layer, ideal for finding shortest connections and assessing zone accessibility from a starting point.

DFS (Depth-First Search): Explores deeply into each branch, useful for cycle detection and alternate path exploration when primary routes are congested.

27void SmartCityTraffic::addRoad(int u, int v) {

36// Breadth First Search: Finds shortest path-like exploration


37void SmartCityTraffic::runBFS(int start) {


38    vector<bool> visited(numZones, false);


39    queue<int> q;


40


41    visited[start] = true;


42    q.push(start);


43


44    cout << "\nBFS Traversal from " << zoneNames[start] << ":\n";


45


46    while (!q.empty()) {


47        int u = q.front();


48        q.pop();


49        cout << zoneNames[u] << " -> ";


50


51        for (int v : roadNetwork[u]) {


52            if (!visited[v]) {


53                visited[v] = true;


54                q.push(v);


55            }


56        }


57    }


58    cout << "END\n";


59}


60


61// Depth First Search: Useful for cycle detection or exploring deep alternate paths


62void SmartCityTraffic::runDFS(int start) {


63    vector<bool> visited(numZones, false);


64    cout << "\nDFS Traversal from " << zoneNames[start] << ":\n";


65    dfsHelper(start, visited);


66    cout << "END\n";


67}


68


69void SmartCityTraffic::dfsHelper(int node, vector<bool> &visited) {


70    visited[node] = true;


71    cout << zoneNames[node] << " -> ";


72    for (int neighbor : roadNetwork[node]) {


73        if (!visited[neighbor]) {


74            dfsHelper(neighbor, visited);


75        }


76    }


77}
