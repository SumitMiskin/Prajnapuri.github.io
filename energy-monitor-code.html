Problem Statement

Managing a smart energy grid requires real-time monitoring of power consumption across various sectors. We need to efficiently calculate the total power usage for specific districts (ranges) and update readings as they change instantly.

Solution Approach

We use a **Segment Tree** data structure to handle range measurements and point updates efficiently. The tree stores aggregate power usage at each node, allowing for O(log n) complexity for both updating a meter's reading and querying the total consumption of a range of meters.

This allows the grid control center to instantly spot high-demand areas and adjust distribution accordingly.

C++ Implementation
1#include <iostream>


2#include <vector>


3using namespace std;


4


5// Segment Tree for Energy Grid


6class EnergyMonitor {


7    vector<int> tree;


8    int n;


9


10    void build(const vector<int>& data, int node, int start, int end) {


11        if (start == end) {


12            tree[node] = data[start];


13        } else {


14            int mid = (start + end) / 2;


15            build(data, 2 * node, start, mid);


16            build(data, 2 * node + 1, mid + 1, end);


17            tree[node] = tree[2 * node] + tree[2 * node + 1];


18        }


19    }


20


21    void updateRecursive(int node, int start, int end, int idx, int val) {


22        if (start == end) {


23            tree[node] = val;


24        } else {


25            int mid = (start + end) / 2;


26            if (start <= idx && idx <= mid)


27                updateRecursive(2 * node, start, mid, idx, val);


28            else


29                updateRecursive(2 * node + 1, mid + 1, end, idx, val);


30            tree[node] = tree[2 * node] + tree[2 * node + 1];


31        }


32    }


33


34    int queryRecursive(int node, int start, int end, int l, int r) {


35        if (r < start || end < l) return 0;


36        if (l <= start && end <= r) return tree[node];


37        int mid = (start + end) / 2;


38        return queryRecursive(2 * node, start, mid, l, r) +


39               queryRecursive(2 * node + 1, mid + 1, end, l, r);


40    }


41


42public:


43    EnergyMonitor(const vector<int>& data) {


44        n = data.size();


45        tree.resize(4 * n);


46        build(data, 1, 0, n - 1);


47    }


48


49    void update(int idx, int val) {


50        updateRecursive(1, 0, n - 1, idx, val);


51    }


52


53    int query(int l, int r) {


54        return queryRecursive(1, 0, n - 1, l, r);


55    }


56};


57


58int main() {


59    // Initial readings: Zone 0 to 5


60    vector<int> grid = {10, 20, 15, 30, 25, 40};


61    EnergyMonitor em(grid);


62


63    // Query total consumption in zones 0-5


64    cout << "Total Usage (0-5): " << em.query(0, 5) << " kWh\n";


65    // Query total consumption in zones 1-3


66    cout << "Usage in Sector B (1-3): " << em.query(1, 3) << " kWh\n";


67


68    // Update Zone 2 to 50 kWh


69    cout << "Updating Zone 2 to 50 kWh...\n";


70    em.update(2, 50);


71


72    cout << "New Usage in Sector B (1-3): " << em.query(1, 3) << " kWh\n";


73    return 0;


74}
