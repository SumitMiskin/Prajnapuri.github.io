Problem Statement

grid stability depends on maintaining stored voltage levels within safe limits. We need a way to instantly identify the lowest voltage recorded across any range of checkpoints to detect potential brownouts or weak points in the historical data.

Solution Approach

We use a **Sparse Table** data structure to perform Range Minimum Queries (RMQ) efficiently. By precomputing minima for intervals of power-of-two lengths, we can answer any range minimum query in **O(1)** time.

This is ideal for analyzing static logs where voltage data is collected and we need to run multiple diagnostic queries rapidly.

C++ Implementation
1#include <iostream>


2#include <vector>


3#include <cmath>


4using namespace std;


5


6class EnergyGridSparseTable {


7private:


8    vector<vector<int>> st;


9    vector<int> logTable;


10


11public:


12    EnergyGridSparseTable(const vector<int>& voltage) {


13        int n = voltage.size();


14        int maxLog = log2(n) + 1;


15


16        logTable.resize(n + 1);


17        for (int i = 2; i <= n; i++)


18            logTable[i] = logTable[i / 2] + 1;


19


20        st.resize(n, vector<int>(maxLog));


21        for (int i = 0; i < n; i++)


22            st[i][0] = voltage[i];


23


24        for (int j = 1; j < maxLog; j++) {


25            for (int i = 0; i + (1 << j) <= n; i++) {


26                st[i][j] = min(st[i][j - 1], st[i + (1 << (j - 1))][j - 1]);


27            }


28        }


29    }


30


31    // Returns the minimum voltage in range [l, r]


32    int query(int l, int r) {


33        int len = r - l + 1;


34        int j = logTable[len];


35        return min(st[l][j], st[r - (1 << j) + 1][j]);


36    }


37};


38


39int main() {


40    int n;


41    cout << "Enter number of checkpoints: ";


42    cin >> n;


43


44    vector<int> voltage(n);


45    cout << "Enter voltage readings at each checkpoint:\n";


46    for (int i = 0; i < n; i++) {


47        cin >> voltage[i];


48    }


49


50    EnergyGridSparseTable grid(voltage);


51


52    int q;


53    cout << "Enter number of voltage queries: ";


54    cin >> q;


55


56    while (q--) {


57        int l, r;


58        cout << "Enter range of checkpoints (1-based indexing): ";


59        cin >> l >> r;


60        l--; r--; // convert to 0-based indexing


61        cout << "Minimum voltage in range [" << l + 1 << ", " << r + 1 << "] is: " << grid.query(l, r) << " volts\n";


62    }


63


64    return 0;


65}
