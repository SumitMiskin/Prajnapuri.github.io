Problem Statement

Efficiently store and retrieve real-time traffic sensor data (vehicle counts, signal status, congestion levels) indexed by sensor ID for O(1) average-case lookup and update operations.

Solution Approach

Hash Table with Chaining: Uses a hash function to map sensor IDs to buckets in a fixed-size table. Collisions are resolved via linked-list chaining. Provides fast insert, search, and delete operations for managing distributed traffic monitoring sensors across the city.

5const int TABLE_SIZE = 10;

7// Node structure for storing traffic sensor data


8struct Node {


9    int sensorID;         // Unique sensor or location ID


10    string trafficData;   // Example: "25 vehicles", "Red Signal", "Congested"


11    Node* next;


12


13    Node(int id, string data) {


14        sensorID = id;


15        trafficData = data;


16        next = NULL;


17    }


18};


19


20class TrafficHashTable {


21private:


22    Node* table[TABLE_SIZE];


23


24    // Simple hash function


25    int hashFunction(int key) {


26        return key % TABLE_SIZE;


27    }


28


29public:


30    TrafficHashTable() {


31        for (int i = 0; i < TABLE_SIZE; ++i)


32            table[i] = NULL;


33    }


34


35    // Insert or update sensor data


36    void insert(int sensorID, const string& data) {


37        int hashKey = hashFunction(sensorID);


38        Node* newNode = new Node(sensorID, data);


39


40        if (table[hashKey] == NULL) {


41            table[hashKey] = newNode;


42        } else {


43            Node* temp = table[hashKey];


44            while (temp->next != NULL) {


45                if (temp->sensorID == sensorID) {


46                    temp->trafficData = data;


47                    delete newNode;


48                    return;


49                }


50                temp = temp->next;


51            }


52            if (temp->sensorID == sensorID) {


53                temp->trafficData = data;


54                delete newNode;


55                return;


56            }


57            temp->next = newNode;


58        }


59    }


60


61    // Search for traffic data using sensor ID


62    void search(int sensorID) {


63        int hashKey = hashFunction(sensorID);


64        Node* temp = table[hashKey];


65        while (temp != NULL) {


66            if (temp->sensorID == sensorID) {


67                cout << "Traffic data at sensor " << sensorID << ": " << temp->trafficData << endl;


68                return;


69            }


70            temp = temp->next;


71        }


72        cout << "Sensor ID " << sensorID << " not found.\n";


73    }


74


75    // Remove sensor data


76    void remove(int sensorID) {


77        int hashKey = hashFunction(sensorID);


78        Node* temp = table[hashKey];


79        Node* prev = NULL;


80        while (temp != NULL && temp->sensorID != sensorID) {


81            prev = temp;


82            temp = temp->next;


83        }


84        if (temp == NULL) {


85            cout << "Sensor ID " << sensorID << " not found.\n";


86            return;


87        }


88        if (prev == NULL) {


89            table[hashKey] = temp->next;


90        } else {


91            prev->next = temp->next;


92        }


93        delete temp;


94        cout << "Sensor ID " << sensorID << " removed.\n";


95    }


96};
