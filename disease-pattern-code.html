Problem Statement

Analyzing large volumes of unstructured medical text (e.g., patient history notes) for specific disease keywords is essential for early diagnosis and research. Standard search methods can be slow when processing thousands of documents.

Solution Approach

**KMP (Knuth-Morris-Pratt)** allows for linear-time searching of a single pattern by avoiding redundant comparisons. **Rabin-Karp** uses hashing to efficiently search for multiple patterns (e.g., a list of diseases) simultaneously, making it ideal for detecting various conditions in one pass.

6// ----------------- KMP ALGORITHM -----------------


7void computeLPSArray(string pattern, vector<int>& lps) {


8    int length = 0, i = 1;


9    lps[0] = 0;


10


11    while (i < pattern.size()) {


12        if (pattern[i] == pattern[length]) {


13            length++;


14            lps[i++] = length;


15        } else {


16            if (length != 0)


17                length = lps[length - 1];


18            else


19                lps[i++] = 0;


20        }


21    }


22}

24void KMPSearch(string text, string pattern) {


25    int n = text.size();


26    int m = pattern.size();


27    vector<int> lps(m);


28


29    computeLPSArray(pattern, lps);


30


31    int i = 0, j = 0;


32    while (i < n) {


33        if (pattern[j] == text[i]) {


34            i++, j++;


35        }


36        if (j == m) {


37            cout << "Disease pattern found at index " << i - j << endl;


38            j = lps[j - 1];


39        } else if (i < n && pattern[j] != text[i]) {


40            if (j != 0) j = lps[j - 1];


41            else i++;


42        }


43    }


44}

46// ----------------- RABIN-KARP ALGORITHM -----------------


47void RabinKarp(string text, vector<string> patterns, int d, int q) {


48    int n = text.size();


49


50    for (string pattern : patterns) {


51        int m = pattern.size();


52        int h = 1;


53        for (int i = 0; i < m - 1; i++)


54            h = (h * d) % q;


55


56        int p = 0, t = 0;


57        for (int i = 0; i < m; i++) {


58            p = (d * p + pattern[i]) % q;


59            t = (d * t + text[i]) % q;


60        }


61


62        for (int s = 0; s <= n - m; s++) {


63            if (p == t) {


64                bool match = true;


65                for (int i = 0; i < m; i++) {


66                    if (text[s + i] != pattern[i]) {


67                        match = false;


68                        break;


69                    }


70                }


71                if (match) {


72                    cout << "Disease \"" << pattern << "\" found at index " << s << endl;


73                }


74            }


75


76            if (s < n - m) {


77                t = (d * (t - text[s] * h) + text[s + m]) % q;


78                if (t < 0) t += q;


79            }


80        }


81    }


82}
