Problem Statement

Efficiently search for flagged vehicle license plates in ANPR (Automatic Number Plate Recognition) logs to identify violators, stolen vehicles, or vehicles of interest in real-time traffic monitoring systems.

Solution Approach

Rabin-Karp Algorithm: Uses rolling hash to find pattern matches with average-case O(n+m) time complexity. Computes hash values for the target plate and sliding windows of the log stream, enabling fast detection of vehicle plates without scanning character-by-character.

6// Rabin-Karp pattern matching adapted for ANPR


7void searchPlate(const string& logStream, const string& targetPlate, int base, int primeMod)


8{


9    int logLen = logStream.size();


10    int plateLen = targetPlate.size();


11


12    int hashMultiplier = 1;


13    for (int i = 0; i < plateLen - 1; i++)


14        hashMultiplier = (hashMultiplier * base) % primeMod;


15


16    int patternHash = 0, textHash = 0;


17


18    // Initial hash values for pattern and first window of text


19    for (int i = 0; i < plateLen; i++) {


20        patternHash = (base * patternHash + targetPlate[i]) % primeMod;


21        textHash = (base * textHash + logStream[i]) % primeMod;


22    }


23


24    // Slide over the log stream


25    for (int s = 0; s <= logLen - plateLen; s++) {


26        // Check hash match first


27        if (patternHash == textHash) {


28            bool matched = true;


29            for (int i = 0; i < plateLen; i++) {


30                if (logStream[s + i] != targetPlate[i]) {


31                    matched = false;


32                    break;


33                }


34            }


35            if (matched)


36                cout << "Vehicle plate '" << targetPlate << "' found at position " << s << " in the stream.\n";


37        }


38


39        // Update hash for next window


40        if (s < logLen - plateLen) {


41            textHash = (base * (textHash - logStream[s] * hashMultiplier) + logStream[s + plateLen]) % primeMod;


42            if (textHash < 0)


43                textHash += primeMod;


44        }


45    }


46}
