Problem Statement

Smart meters generate massive amounts of time-stamped usage data throughout the day. To support fast billing inquiries and historical analysis, we need a way to store these records such that we can quickly search for a specific timestamp, even as new data points are continually added.

Solution Approach

We use an **AVL Tree** (Assumption-less Balanced Binary Search Tree) to store meter readings indexed by timestamp. Since AVL trees self-balance after every insertion, they guarantee **O(log n)** time complexity for search, insertion, and deletion.

This ensures that even with millions of records, retrieving a specific reading remains extremely fast and consistent.

C++ Implementation
1#include <iostream>


2using namespace std;


3


4struct MeterData {


5    int timestamp;  // or meter ID


6    float reading;


7};


8


9struct AVLNode {


10    MeterData data;


11    AVLNode* left;


12    AVLNode* right;


13    int height;


14


15    AVLNode(MeterData val) {


16        data = val;


17        left = right = nullptr;


18        height = 1;


19    }


20};


21


22int getHeight(AVLNode* node) {


23    return node ? node->height : 0;


24}


25


26int getBalance(AVLNode* node) {


27    return node ? getHeight(node->left) - getHeight(node->right) : 0;


28}


29


30AVLNode* rightRotate(AVLNode* y) {


31    AVLNode* x = y->left;


32    AVLNode* T2 = x->right;


33    x->right = y;


34    y->left = T2;


35


36    y->height = max(getHeight(y->left), getHeight(y->right)) + 1;


37    x->height = max(getHeight(x->left), getHeight(x->right)) + 1;


38


39    return x;


40}


41


42AVLNode* leftRotate(AVLNode* x) {


43    AVLNode* y = x->right;


44    AVLNode* T2 = y->left;


45    y->left = x;


46    x->right = T2;


47


48    x->height = max(getHeight(x->left), getHeight(x->right)) + 1;


49    y->height = max(getHeight(y->left), getHeight(y->right)) + 1;


50


51    return y;


52}


53


54AVLNode* insert(AVLNode* node, MeterData data) {


55    if (!node)


56        return new AVLNode(data);


57


58    if (data.timestamp < node->data.timestamp)


59        node->left = insert(node->left, data);


60    else if (data.timestamp > node->data.timestamp)


61        node->right = insert(node->right, data);


62    else


63        return node; // duplicate timestamp not allowed


64


65    node->height = 1 + max(getHeight(node->left), getHeight(node->right));


66


67    int balance = getBalance(node);


68


69    if (balance > 1 && data.timestamp < node->left->data.timestamp)


70        return rightRotate(node);


71


72    if (balance < -1 && data.timestamp > node->right->data.timestamp)


73        return leftRotate(node);


74


75    if (balance > 1 && data.timestamp > node->left->data.timestamp) {


76        node->left = leftRotate(node->left);


77        return rightRotate(node);


78    }


79


80    if (balance < -1 && data.timestamp < node->right->data.timestamp) {


81        node->right = rightRotate(node->right);


82        return leftRotate(node);


83    }


84


85    return node;


86}

88void inOrder(AVLNode* root) {


89    if (root) {


90        inOrder(root->left);


91        cout << "Timestamp: " << root->data.timestamp


92             << " | Reading: " << root->data.reading << " kWh" << endl;


93        inOrder(root->right);


94    }


95}


96


97AVLNode* search(AVLNode* root, int timestamp) {


98    if (!root || root->data.timestamp == timestamp)


99        return root;


100    if (timestamp < root->data.timestamp)


101        return search(root->left, timestamp);


102    return search(root->right, timestamp);


103}
