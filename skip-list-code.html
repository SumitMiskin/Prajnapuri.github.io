Problem Statement

Smart meter networks require a scalable way to maintain an ordered list of active meter IDs for fast lookups, insertions, and validation. As the network expands, we need a data structure that offers efficient search performance similar to balanced trees but is easier to implement and maintain.

Solution Approach

A **Skip List** is a probabilistic data structure that allows **O(log n)** search complexity essentially by maintaining multiple layers of linked lists. It "skips" over elements to find the target quickly, providing a highly efficient alternative to binary search trees for dynamic datasets in the energy grid.

6#define MAX_LEVEL 6


7float P = 0.5;

9struct Node {


10    int key;


11    Node** forward;


12    Node(int key, int level) {


13        this->key = key;


14        forward = new Node*[level + 1];


15        for (int i = 0; i <= level; i++) {


16            forward[i] = nullptr;


17        }


18    }


19};


20


21class SkipList {


22    int level;


23    Node* header;


24public:


25    SkipList() {


26        level = 0;


27        header = new Node(-1, MAX_LEVEL);  // Dummy header


28    }


29


30    int randomLevel() {


31        int lvl = 0;


32        while (((float)rand() / RAND_MAX) < P && lvl < MAX_LEVEL)


33            lvl++;


34        return lvl;


35    }


36


37    void insert(int key) {


38        Node* current = header;


39        Node* update[MAX_LEVEL + 1];


40        for (int i = level; i >= 0; i--) {


41            while (current->forward[i] && current->forward[i]->key < key)


42                current = current->forward[i];


43            update[i] = current;


44        }


45


46        current = current->forward[0];


47


48        if (!current || current->key != key) {


49            int rlevel = randomLevel();


50


51            if (rlevel > level) {


52                for (int i = level + 1; i <= rlevel; i++) {


53                    update[i] = header;


54                }


55                level = rlevel;


56            }


57


58            Node* newNode = new Node(key, rlevel);


59            for (int i = 0; i <= rlevel; i++) {


60                newNode->forward[i] = update[i]->forward[i];


61                update[i]->forward[i] = newNode;


62            }


63            cout << "Inserted smart meter ID: " << key << endl;


64        }


65    }


66


67    void search(int key) {


68        Node* current = header;


69        for (int i = level; i >= 0; i--) {


70            while (current->forward[i] && current->forward[i]->key < key)


71                current = current->forward[i];


72        }


73        current = current->forward[0];


74        if (current && current->key == key)


75            cout << "Smart meter ID " << key << " found.\n";


76        else


77            cout << "Smart meter ID " << key << " not found.\n";


78    }


79


80    void display() {


81        cout << "\nSkip List (smart meter IDs):\n";


82        for (int i = 0; i <= level; i++) {


83            Node* node = header->forward[i];


84            cout << "Level " << i << ": ";


85            while (node) {


86                cout << node->key << " ";


87                node = node->forward[i];


88            }


89            cout << "\n";


90        }


91    }


92};
