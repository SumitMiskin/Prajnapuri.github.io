Problem Statement

During an epidemic, tracking the spread of infection through contact tracing and identifying affected hospital wards is critical. We need to efficiently group connected patients (clusters) and traverse the hospital network to isolate affected zones.

Solution Approach

**Union-Find (Disjoint Set)** efficiently groups patients into infection clusters based on contact history. **BFS (Breadth-First Search)** traverses the hospital ward graph to identify all zones reachable from an infected source, enabling rapid containment strategies.

6// ------------ UNION-FIND (Disjoint Set for Patient Clustering) ------------


7class UnionFind {


8    vector<int> parent;


9


10public:


11    UnionFind(int n) {


12        parent.resize(n);


13        for(int i = 0; i < n; i++) parent[i] = i;


14    }


15


16    int find(int u) {


17        if (parent[u] != u)


18            parent[u] = find(parent[u]); // Path compression


19        return parent[u];


20    }


21


22    void unite(int u, int v) {


23        int pu = find(u);


24        int pv = find(v);


25        if (pu != pv) {


26            parent[pu] = pv;


27        }


28    }


29


30    void printClusters(int n) {


31        cout << "\nPatient Infection Clusters:\n";


32        for (int i = 0; i < n; i++) {


33            cout << "Patient " << i << " belongs to cluster " << find(i) << endl;


34        }


35    }


36};

38// ------------ BFS for Affected Ward Segmentation ------------


39void bfsAffectedWards(vector<vector<int>>& hospital, int start, vector<bool>& visited) {


40    queue<int> q;


41    q.push(start);


42    visited[start] = true;


43


44    cout << "\nBFS Traversal from Ward " << start << ":\nAffected Wards: ";


45    while (!q.empty()) {


46        int ward = q.front();


47        q.pop();


48        cout << ward << " ";


49        for (int neighbor : hospital[ward]) {


50            if (!visited[neighbor]) {


51                visited[neighbor] = true;


52                q.push(neighbor);


53            }


54        }


55    }


56    cout << endl;


57}
