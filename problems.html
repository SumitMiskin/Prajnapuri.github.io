Optimized way for water supply
Problem Statement

Providing tap water supply for each corner of the city with minimal pipe-line connection to reduce the cost for the project is bit difficult. We need to implement the pipe-line with cost and can optimize the cost and budget. We need to find the shortest path and implement these pipes in efficient manner.

Solution Approach

To solve the pipe-line problem we are using minimum spanning tree: Prim’s algorithm. Adding vertices in each corner to cover all the area of our city and the edges are connecting each node for proper implementation.

C++ Implementation
#include <bits/stdc++.h>
#define V 100          // max number of junctions
#define INF 1000000
// Find vertex with minimum key value not yet in MST
int minKey(int key[], int inMST[], int n) {
    int min = INF, minIndex = -1;
    for (int v = 0; v < n; v++) {
        if (!inMST[v] && key[v] < min) {
            min = key[v];
            minIndex = v;
        }
    }
    return minIndex;
}
// Prim's algorithm: builds MST for given cost matrix
void primMST(int graph[V][V], int n) {
    int parent[V];   // parent[i] = parent of i in MST
    int key[V];      // minimum edge weight to connect i
    int inMST[V];    // 1 if vertex included in MST
    for (int i = 0; i < n; i++) {
        key[i] = INF;
        inMST[i] = 0;
    }
    // Start from junction 0 (can choose any)
    key[0] = 0;
    parent[0] = -1;
    // MST will have n-1 edges
    for (int count = 0; count < n - 1; count++) {
        int u = minKey(key, inMST, n);
        inMST[u] = 1;
        // Update key values of adjacent vertices
        for (int v = 0; v < n; v++) {
            if (graph[u][v] != INF && !inMST[v] && graph[u][v] < key[v]) {
                parent[v] = u;
                key[v] = graph[u][v];
            }
        }
    }
    // Print chosen pipes
    int totalCost = 0;
    printf("Selected pipes (MST edges): ");
    for (int i = 1; i < n; i++) {
        printf("Junction %d -- Junction %d  cost = %d", parent[i], i, graph[i][parent[i]]);
        totalCost += graph[i][parent[i]];
    }
    printf("Total pipeline cost (MST) = %d", totalCost);
}

int main() {
    int n;
    int graph[V][V];
    printf("Enter number of junctions: ");
    scanf("%d", &n);
    printf("Enter cost matrix (%d x %d), use %d if no direct pipe:",  n, n, INF);
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            scanf("%d", &graph[i][j]);
        }
    }
    primMST(graph, n);
    return 0;
}
Soil Moisture Monitoring
Problem Statement

In large agricultural fields, soil moisture sensors are placed at multiple locations to measure water content. Farmers often need to know:

  • Moisture level of a specific area
  • Average or total moisture in a range of plots
  • Updated moisture after irrigation or rainfall

If we check moisture one sensor at a time, it becomes:

  • Slow
  • Inefficient
  • Not suitable for large farms
Solution Approach

A Segment Tree is a special data structure used to:

  • Store data in segments (ranges)
  • Answer range queries efficiently
  • Update values quickly

Instead of checking each sensor individually, the segment tree divides the field into smaller segments and stores their moisture data in a tree structure. This is where a Segment Tree becomes useful.

Implementation
#include iostream
using namespace std;
class SegmentTree {
    int *tree, *arr, n;
public:
    SegmentTree(int input[], int size) {
        n = size;
        arr = input;
        tree = new int[4 * n];
        build(0, 0, n - 1);
    }
    void build(int node, int start, int end) {
        if (start == end) {
            tree[node] = arr[start];
        } else {
            int mid = (start + end) / 2;
            build(2 * node + 1, start, mid);
            build(2 * node + 2, mid + 1, end);
            tree[node] = tree[2 * node + 1] + tree[2 * node + 2];
        }
    }
    int query(int node, int start, int end, int l, int r) {
        if (r < start || l > end)
            return 0;
        if (l <= start && end <= r)
            return tree[node];
        int mid = (start + end) / 2;
        return query(2 * node + 1, start, mid, l, r) +
               query(2 * node + 2, mid + 1, end, l, r);
    }
};

Quick sort:
// Quick Sort
// Sorting the data by quicksort for names, and age
// Input: A subarray A[l...r] of A[0...n-1], defined by its left and right indices l and r
// Output: sorted by names in first condition and reusing it for ages.
#include iostream
void QuickSort(int A[], int l, int r)
{
    if (l < r)
    {
        int s = Partition(A, l, r);
        QuickSort(A, l, s - 1);
        QuickSort(A, s + 1, r);
    }
}
// Partition
// Partitions a subarray by using its first element as a pivot
// Input: A subarray A[l...r] of A[0...n-1], defined by its left and right indices l and r
// (l < r)
// Output: Subarray A[l...r], with split position returned as this function’s value
int Partition(int A[], int l, int r)
{
    int p = A[l];
    int i = l;
    int j = r + 1;
    do
    {
        do
        {
            i = i + 1;
        } while (A[i] >= p);
        do
        {
            j = j - 1;
        } while (A[j] <= p);
        swap(A[i], A[j]);
    } while (i >= j);
    swap(A[i], A[j]);
    swap(A[l], A[j]);
    return j;
}

Drone Delivery Routing
Problem Statement

Providing a sufficient and stable electricity for our city Prajnapuri as it has covered major part as IT sector and electricity plays a major role here. In this project we’re planning to provide a stable 24/7 electricity which is provided by the main Power station of the located in the city (source place).

Solution Approach

To solve the Electricity problem we are using mostly popular algorithm Dijkstra’s algorithm. Adding vertices in each corner to cover all the area of our city and the edges are connecting each node for proper implementation.

Implementation
#include 
using namespace std;
#define MAXN 100
#define INF 999
int main() {
    int n;
    cout << "Enter the number of Vertices of the city Prajnapuri\n";
    cin >> n;
    int cost[MAXN][MAXN];
    cout << "Enter the cost matrix: (Enter 999 for Infinity)\n";
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            cin >> cost[i][j];
        }
    }
    int source;
    cout << "Enter the source\n";
    cin >> source;	//source as the Main power station

    int dist[MAXN], path[MAXN], visited[MAXN];
    for (int i = 0; i < n; i++) {
        dist[i] = cost[source][i];
        path[i] = source;
        visited[i] = 0;
    }
    visited[source] = 1;
    for (int iter = 0; iter < n - 1; iter++) {
        int u = -1;
        int min = INF + 1;
        for (int i = 0; i < n; i++) {
            if (!visited[i] && dist[i] < min) {
                min = dist[i];
                u = i;
            }
        }
        if (u == -1)
            break;
        visited[u] = 1;
        for (int v = 0; v < n; v++) {
            if (!visited[v] && cost[u][v] != INF) {
                if (dist[u] + cost[u][v] < dist[v]) {
                    dist[v] = dist[u] + cost[u][v];
                    path[v] = u;
                }
            }
        }
    }
    cout << "\nVertex   Dist   Path\n";
    for (int i = 0; i < n; i++) {
        cout << i << "        " << dist[i] << "      " << path[i] << "\n";
    }
    return 0;
}
