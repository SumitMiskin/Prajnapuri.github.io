Problem Statement

To balance the grid effectively, we often need to rank transformers, generation units, or customer usage profiles by load. This helps in identifying overloaded components for maintenance or selecting low-demand areas for load shifting.

Solution Approach

No single sorting algorithm fits all scenarios. We use **Insertion Sort** for small datasets (like a local substation's device list) due to its simplicity and low overhead. For larger datasets (city-wide logs), we employ **Merge Sort** or **Quick Sort** (`O(n log n)`) to handle massive sorting tasks efficiently.

5// ------------------ Insertion Sort ------------------


6void insertionSort(vector<pair<string, int>>& data) {


7    for (int i = 1; i < data.size(); ++i) {


8        auto key = data[i];


9        int j = i - 1;


10        while (j >= 0 && data[j].second > key.second) {


11            data[j + 1] = data[j];


12            j--;


13        }


14        data[j + 1] = key;


15    }


16}

18// ------------------ Merge Sort ------------------


19void merge(vector<pair<string, int>>& data, int left, int mid, int right) {


20    vector<pair<string, int>> L(data.begin() + left, data.begin() + mid + 1);


21    vector<pair<string, int>> R(data.begin() + mid + 1, data.begin() + right + 1);


22


23    int i = 0, j = 0, k = left;


24    while (i < L.size() && j < R.size()) {


25        if (L[i].second <= R[j].second)


26            data[k++] = L[i++];


27        else


28            data[k++] = R[j++];


29    }


30    while (i < L.size()) data[k++] = L[i++];


31    while (j < R.size()) data[k++] = R[j++];


32}


33


34void mergeSort(vector<pair<string, int>>& data, int left, int right) {


35    if (left < right) {


36        int mid = (left + right) / 2;


37        mergeSort(data, left, mid);


38        mergeSort(data, mid + 1, right);


39        merge(data, left, mid, right);


40    }


41}

43// ------------------ Quick Sort ------------------


44int partition(vector<pair<string, int>>& data, int low, int high) {


45    int pivot = data[high].second;


46    int i = low - 1;


47    for (int j = low; j < high; ++j) {


48        if (data[j].second < pivot) {


49            i++;


50            swap(data[i], data[j]);


51        }


52    }


53    swap(data[i + 1], data[high]);


54    return i + 1;


55}


56


57void quickSort(vector<pair<string, int>>& data, int low, int high) {


58    if (low < high) {


59        int pi = partition(data, low, high);


60        quickSort(data, low, pi - 1);


61        quickSort(data, pi + 1, high);


62    }


63}
