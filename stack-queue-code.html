Problem Statement

Online learning platforms need to support undo operations for text editing and manage student waitlists for popular courses efficiently.

Solution Approach

We implement a **Stack** (LIFO) to track user actions for 'Undo' functionality and a **Queue** (FIFO) to manage course enrollment waitlists fairly, ensuring the first student to join is the first to get a seat.

C++ Implementation
1#include <iostream>


2#include <string>


3using namespace std;


4


5#define MAX 100


6


7// ---------- Stack for Undo Actions ----------


8class UndoStack {


9    string actions[MAX];


10    int top;


11


12public:


13    UndoStack() { top = -1; }


14


15    void push(string action) {


16        if (top >= MAX - 1)


17            cout << "Undo stack is full!" << endl;


18        else


19            actions[++top] = action;


20    }


21


22    void undo() {


23        if (top == -1)


24            cout << "Nothing to undo!" << endl;


25        else


26            cout << "Undo action: " << actions[top--] << endl;


27    }


28


29    void show() {


30        if (top == -1) {


31            cout << "No actions recorded.\n";


32            return;


33        }


34        cout << "Undo stack (latest first): ";


35        for (int i = top; i >= 0; i--)


36            cout << actions[i] << " ";


37        cout << endl;


38    }


39};


40


41// ---------- Queue for Course Signup Waitlist ----------


42class CourseQueue {


43    string waitlist[MAX];


44    int front, rear;


45


46public:


47    CourseQueue() {


48        front = rear = -1;


49    }


50


51    void enqueue(string studentName) {


52        if (rear == MAX - 1)


53            cout << "Waitlist full!" << endl;


54        else {


55            if (front == -1) front = 0;


56            waitlist[++rear] = studentName;


57        }


58    }


59


60    void dequeue() {


61        if (front == -1 || front > rear)


62            cout << "No students in waitlist." << endl;


63        else


64            cout << "Student removed from waitlist: " << waitlist[front++] << endl;


65    }


66


67    void display() {


68        if (front == -1 || front > rear) {


69            cout << "Waitlist is empty." << endl;


70            return;


71        }


72


73        cout << "Current course waitlist: ";


74        for (int i = front; i <= rear; i++)


75            cout << waitlist[i] << " ";


76        cout << endl;


77    }


78};
