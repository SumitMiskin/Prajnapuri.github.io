Problem Statement

Smart grid systems generate massive amounts of log data, containing critical events like "overload", "fault", or "outage". To quickly react to emergencies, we need to scan these logs in real-time to find specific keywords or error patterns.

Solution Approach

The **Boyer-Moore String Search** algorithm is an extremely efficient pattern matching algorithm. It skips sections of the text that cannot possibly contain a match, allowing it to search large log files much faster than a standard linear scan, often achieving sub-linear time complexity.

8// Preprocess the bad character heuristic


9void badCharHeuristic(const string& pattern, vector<int>& badChar) {


10    for (int i = 0; i < NO_OF_CHARS; i++)


11        badChar[i] = -1;


12    for (int i = 0; i < pattern.length(); i++)


13        badChar[(int) pattern[i]] = i;


14}

16// Boyer-Moore Search Function


17void boyerMooreSearch(const string& text, const string& pattern) {


18    int m = pattern.size();


19    int n = text.size();


20


21    vector<int> badChar(NO_OF_CHARS);


22    badCharHeuristic(pattern, badChar);


23


24    int shift = 0;


25    while (shift <= (n - m)) {


26        int j = m - 1;


27


28        while (j >= 0 && pattern[j] == text[shift + j])


29            j--;


30


31        if (j < 0) {


32            cout << "Pattern found at index " << shift << endl;


33            shift += (shift + m < n) ? m - badChar[text[shift + m]] : 1;


34        } else {


35            shift += max(1, j - badChar[text[shift + j]]);


36        }


37    }


38}
