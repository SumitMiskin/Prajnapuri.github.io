7public:

19void SmartTransportation::initialize(int timeMatrix[4][4], int start) {

    
20    for (int i = 0; i < 4; i++) {

    
21        previous[i] = start;

    
22        travelTime[i] = (timeMatrix[start][i] == 0) ? 10000000 : timeMatrix[start][i];

    
23        visited[i] = 0;

    
24    }

    
25    visited[start] = 1;

    
26    travelTime[start] = 0;

    
27}

    
28

    
29int SmartTransportation::findMinimum() {

    
30    int minimum = 10000000;

    
31    int minIndex = -1;

    
32    for (int i = 0; i < 4; i++) {

    
33        if (!visited[i] && travelTime[i] < minimum) {

    
34            minimum = travelTime[i];

    
35            minIndex = i;

    
36        }

    
37    }

    
38    return minIndex;

    
39}

    
40

    
41void SmartTransportation::update(int timeMatrix[4][4]) {

    
42    int u = findMinimum();

    
43    if (u == -1)

    
44        return;

    
45    visited[u] = 1;

    
46    for (int i = 0; i < 4; i++) {

    
47        if (!visited[i]) {

    
48            if (travelTime[u] + timeMatrix[u][i] < travelTime[i]) {

    
49                travelTime[i] = travelTime[u] + timeMatrix[u][i];

    
50                previous[i] = u;

    
51            }

    
52        }

    
53    }

    
54}

    
55

    
56void SmartTransportation::computeShortestPaths(int timeMatrix[4][4], int start) {

    
57    initialize(timeMatrix, start);

    
58    for (int i = 0; i < 3; i++)

    
59        update(timeMatrix);

    
60}

    
61

    
62void SmartTransportation::displayPaths(int start, string locationNames[4]) {

    
63    for (int i = 0; i < 4; i++) {

    
64        if (i != start) {

    
65            cout << "Fastest route from " << locationNames[start] << " to " << locationNames[i]

    
66                 << " takes " << travelTime[i] << " minutes via: ";

    
67            int j = i;

    
68            cout << locationNames[j];

    
69            while (previous[j] != start) {

    
70                j = previous[j];

    
71                cout << " <- " << locationNames[j];

    
72            }

    
73            cout << " <- " << locationNames[start] << endl;

    
74        }

    
75    }

    
76}
