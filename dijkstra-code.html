1// Ride-Sharing Dispatch Demo
2// Algorithms: BFS, DFS, Dijkstra, Bellman-Ford
3// Data structures: queue (request buffer), priority_queue (min-heap for closest driver),
4// unordered_map (hashing for fast driver lookup)
5
6#include <bits/stdc++.h>
7using namespace std;
8
9// ---------------------- Shared Graph Structure --------------------
10// Nodes can represent map intersections or zones.
11// Edges represent roads between them.
12struct Edge {
13    int u, v;
14    int w; // travel time or distance
15};
16
17class Graph {
18public:
19    int V;
20    vector<vector<int>> adjList;               // for BFS/DFS
21    vector<vector<pair<int,int>>> adjWeighted;   // for Dijkstra
22    vector<Edge> edges;                           // for Bellman-Ford
23
24    Graph(int n = 0) {
25        V = n;
26        adjList.assign(V, {});
27        adjWeighted.assign(V, {});
28    }
29    void resize(int n) {
30        V = n;
31        adjList.assign(V, {});
32        adjWeighted.assign(V, {});
33        edges.clear();
34    }
35};
